+// 根据周期计算优先级，周期越短优先级越高
+// 用于 RMS (Rate Monotonic Scheduling) 和 DMS (Deadline Monotonic Scheduling)
+pub fn period_to_priority(period_ms: f64) -> i32 {
+    let k: f64 = 10.0;
